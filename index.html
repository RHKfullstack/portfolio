class ChatController extends Controller
{
    use HandlesReasoningFlow;
    use HandlesImageContext;
    use HandlesFileAttachments;
    use HandlesDirectSSEStream;

    public function stream(Request $req)
    {
        set_time_limit(600);

        // Rate limiting per subscription plan
        $plan = auth()->user()->subscription_plan;
        $maxPerHour = app(SubscriptionPlanService::class)
            ->getFeature($plan, 'max_messages_per_hour');

        // Validate & route to AI model
        return $this->processStream($req, fn() =>
            DeepSeekService::chat(
                model: 'deepseek-v4-pro',

            )
        );
    }

/** Word-level streaming animation */
function AnimatedText({ children }) {
    if (typeof children !== 'string') {
        if (Array.isArray(children)) {
            return children.map((child, i) => {
                if (typeof child === 'string') {
                    return <AnimatedText key={i}>{child}</AnimatedText>;
                }
                return <React.Fragment key={i}>{child}</React.Fragment>;
            });
        }
        return children;
    }

    const parts = children.split(/(\s+)/);
    return parts.map((part, i) =>
        <span key={i} className="stream-word">{part}</span>
    );
}
class DeepSeekService
{
    protected const BASE_URL = 'https://api.deepseek.com/v1';
    protected const CONTEXT_LIMIT = 1_000_000;

    public static function chat(array $config): Stream
    {
        $response = Http::timeout(120)
            ->withHeaders([
                'Authorization' => 'Bearer ' . config('deepseek.key'),
                'Content-Type' => 'application/json',
            ])
            ->send('POST', self::BASE_URL . '/chat/completions', [
                'stream' => true,
                'body' => json_encode([
                    'model' => $config['model'],
                    'messages' => $config['messages'],
                    'max_tokens' => 4096,
                    'temperature' => 0.7,
                ]),
            ]);

        return new SSEStream($response->toPsrResponse());
    }
}
class ProcessServerOwnedTextRun implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue,
        Queueable, SerializesModels;

    public function __construct(
        protected ChatGenerationRun $run,
        protected array $payload,
        protected int $userId,
    ) {}

    public function handle(): void
    {
        $service = app(TextModelRouterService::class);

        $result = $service->route(
            model: $this->payload['model'],
            prompt: $this->payload['message'],
            stream: false,
        );

        $this->run->saveResponse($result);
        BroadcastRunComplete::dispatch($this->run);
    }
}
